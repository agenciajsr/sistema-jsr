5. **Criar src/lib/meta/schemas.ts** — Zod schemas para validar TODA resposta da Meta API:

```typescript
import { z } from 'zod'

// Schema para um item de acao (actions array)
export const metaActionSchema = z.object({
  action_type: z.string(),
  value: z.string(),
})

// Schema para uma conta de anuncio retornada pelo endpoint /owned_ad_accounts ou /client_ad_accounts
export const metaAdAccountSchema = z.object({
  id: z.string(), // formato "act_123456789"
  name: z.string(),
  account_status: z.number(),
  currency: z.string().optional().default('BRL'),
})

export const metaAdAccountsResponseSchema = z.object({
  data: z.array(metaAdAccountSchema),
  paging: z.object({
    cursors: z.object({
      before: z.string().optional(),
      after: z.string().optional(),
    }).optional(),
    next: z.string().optional(),
  }).optional(),
})

// Schema para um insight de campanha
export const metaInsightSchema = z.object({
  campaign_id: z.string(),
  campaign_name: z.string(),
  spend: z.string().default('0'),
  impressions: z.string().default('0'),
  clicks: z.string().default('0'),
  reach: z.string().default('0'),
  cpc: z.string().optional(),
  cpm: z.string().optional(),
  ctr: z.string().optional(),
  actions: z.array(metaActionSchema).optional().default([]),
  date_start: z.string(),
  date_stop: z.string(),
})

export const metaInsightsResponseSchema = z.object({
  data: z.array(metaInsightSchema),
  paging: z.object({
    cursors: z.object({
      before: z.string().optional(),
      after: z.string().optional(),
    }).optional(),
    next: z.string().optional(),
  }).optional(),
})
```

2. **Criar src/lib/inngest/functions/sync-meta-ads.ts** — funcao Inngest:

- Importar inngest client, db, schema (adAccounts, campaignInsights), meta client (fetchMetaAdAccounts, fetchCampaignInsights)
- Exportar `syncMetaAds = inngest.createFunction(...)`:
  - id: 'sync-meta-ads', name: 'Sincronizar Meta Ads'
  - Triggers: DOIS triggers — cron '0 6 * * *' (todo dia 6h) E event 'meta-ads/sync.requested' (botao manual)
  - Handler async ({ step }):

    a) **step.run('sync-ad-accounts')**: chamar fetchMetaAdAccounts(). Para cada conta retornada:
       - Extrair o ID numerico (remover prefixo "act_" do id)
       - Upsert na tabela ad_accounts: se metaAccountId ja existe, atualizar nome, accountStatus, currency, updatedAt. Se nao existe, inserir. Usar `db.insert(...).onConflictDoUpdate(...)` — para isso, adicionar unique constraint em meta_account_id no schema OU usar sql raw merge. Alternativa mais simples: buscar existente com `db.select().from(adAccounts).where(eq(adAccounts.metaAccountId, id))`, se existe update, se nao insert.

    b) Para cada conta ativa no banco (ativo=true, plataforma='meta'):
       - **step.run(`sync-insights-${account.metaAccountId}`)**: chamar fetchCampaignInsights(account.metaAccountId). Para cada insight retornado, upsert em campaign_insights: se ja existe registro com mesmo (adAccountId, date, campaignId), atualizar; se nao, inserir. Usar sql ON CONFLICT ou logica select+insert/update.
       - **step.sleep(`pause-${account.metaAccountId}`, '2s')**: pausa de 2s entre contas para respeitar rate limits da Meta.

    c) Retornar { synced: N } com numero de contas sincronizadas.
